using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace PracticeC
{
    public delegate int MyDeletgate(int a, int b);

    public class Prac
    {       
        public int Sum(int a,int b)
        {
            return a + b;
        }
        public int Difference(int a,int b)
        {
            return a - b;
        }
        
    }
    internal class Example
    {   
        static void Main(string[] args)
        {
            Prac obj = new Prac();
            MyDeletgate sum = new MyDeletgate(obj.Sum);
            MyDeletgate diff = new MyDeletgate(obj.Difference);
            Console.WriteLine(sum(10,20));
            Console.WriteLine(diff(10, 20));

        }

    }
}
